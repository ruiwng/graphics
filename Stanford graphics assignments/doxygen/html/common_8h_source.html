48  case GL_INVALID_ENUM:
49  return "Invalid enum: An unacceptable value is specified for an enumerated argument. The offending command is ignored and has no other side effect than to set the error flag.";
50  case GL_INVALID_VALUE:
51  return "Invalid value: A numeric argument is out of range. The offending command is ignored and has no other side effect than to set the error flag.";
